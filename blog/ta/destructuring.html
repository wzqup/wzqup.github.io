ES6允许按照一定模式，从数组和对象中提取值，对变量进行赋值。

# 数组的解构

let [a,b,c] = [1,2,3]
// a = 1 b = 2 c = 3

变量的取值由它的位置决定，只要等号两边的模式相同，左边的变量就会被赋予对应的值,一些例子:

let [a, [[b], c]] = [1, [[2], 3]];
// a = 1 b = 2 c = 3

let [a, , b] = [1, 2, 3];
// a = 1 b = 3

let [a, ...numArr] = [1, 2, 3, 4];
// a = 1 ,numArr = [2,3,4];

let [x, y, ...z] = ['a'];
x // "a"
y // undefined
z // []

解构赋值允许指定默认值。 注意，ES6 内部使用严格相等运算符（===），判断一个位置是否有值。所以，只有当一个数组成员严格等于undefined，默认值才会生效。如果一个数组成员是null，默认值就不会生效.

let [x = 1] = [undefined];
x // 1

let [x = 1] = [null];
x // null

如果默认值是一个表达式，那么这个表达式是惰性求值的，即只有在用到的时候，才会求值。

function f() {
  console.log('aaa');
}
let [x = f()] = [a];
// 相当于
let x;
if (a === undefined) {
  x = f();
} else {
  x = a;
}

# 对象的解构

变量必须与属性同名，才能取到正确的值。

let { foo, bar, ddd} = { foo: 'aaa', bar: 'bbb',age:18 };
//相当于
let { foo：foo, bar:bar, ddd:ddd} = { foo: 'aaa', bar: 'bbb',age:18 };
foo // "aaa"
bar // "bbb"
ddd // undefined

如果变量名与属性名不一致，必须写成下面这样。

let obj = { first: 'hello', last: 'world' };
let { first: one, last: two } = obj;
one // 'hello'
two // 'world'
//相当于
let { first: one, last: two } = { first: 'hello', last: 'world'};

上面代码说明，first为匹配的key ,真正需要被赋值的变量是one,值是；'hello'。 再来一个嵌套对象示列

const person = {
  name:'魏大帅'，
  info:{
    age:18,
    sex:'男'
  }
}
const {name，info:{age,sex}} = person;
// console.log(name,age,sex)
// 魏大帅 18 男
// console.log(info)
// error  'info' is not defined
const {name，info} = person;
// console.log(name,info)
// 魏大帅 {age:18,sex:'男'}

对象的解构赋值，可以很方便地将现有对象的方法，赋值到某个变量。

// 例一
let { log, sin, cos } = Math;

// 例二
const { log } = console;
log('hello') // hello

# 函数参数的解构

function add([x, y]){
  return x + y;
}

add([1, 2]); // 3

上面代码中，函数add的参数表面上是一个数组，但在传入参数的那一刻，数组参数就被解构成变量x和y。

const person = {
    name: "魏大帅",
        info: {
            age: 18,
            sex: "男",
        },
    };

function sayHi({ name, info:{age,sex} } = {}){
        let sayText = `my name is${name}, age is ${age},sex is ${sex}`
        console.log(sayText);
    }
sayHi(person);
// my name is魏大帅, age is 18,sex is 男

# 常见用途

// 交换变量的值
let x = 1;
let y = 2;
[x, y] = [y, x];

// 提取JSON数据
let jsonData = {
  id: 42,
  status: "OK",
  data: [867, 5309]
};

let { id, status, data: number } = jsonData;

console.log(id, status, number);
// 42, "OK", [867, 5309]
